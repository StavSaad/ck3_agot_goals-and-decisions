gad_create_d_sea_of_dorne_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	major = yes
	ai_check_interval = 120
	desc = gad_create_d_sea_of_dorne_decision_desc

	is_shown = {
		is_landed = yes
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:gad_create_d_sea_of_dorne_decision
			}
		}
		agot_can_create_duchy_title = yes
		NOT = {
			OR = {
				has_title = title:k_tarth
				any_vassal_or_below = {
					has_title = title:k_tarth
				}
			}
		}
		OR = {
			has_title = title:c_ghaston_grey
			has_title = title:c_tallor_islands
			has_title = title:c_wrathrock
			has_title = title:c_severisles
		}
	}

	is_valid = {
		completely_controls_region = gad_sea_of_dorne_isles
	}

	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
		agot_is_available_for_anything = yes
	}

	cost = {
		gold = 300
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 5
		modifier = {
			has_trait = ambitious
			add = 40
		}
		modifier = {
			has_trait = authoritative
			add = 5
		}
	}

	effect = {
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:gad_create_d_sea_of_dorne_decision
		}
		save_scope_as = founder
		show_as_tooltip = { gad_create_d_sea_of_dorne_scripted_effect = yes }
		trigger_event = { id = gad_title_creation_decisions.0001 }
	}

}

gad_proclaim_narrow_sea_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	major = yes
	desc = gad_proclaim_narrow_sea_decision_desc
	ai_goal	= yes
	ai_check_interval = 120
	ai_potential = { always = yes }

	is_shown = {
		is_independent_ruler = yes
		primary_title.tier = tier_kingdom
		primary_title.title_capital_county = {
			title_province = {
				geographical_region = gad_narrow_sea
			}
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:gad_proclaim_narrow_sea_decision
			}
		}
		NOT = { has_title = title:e_narrow_sea }
		NOT = { has_government = nights_watch_government }
	}

	is_valid = {
		custom_description = {
			text = gad_proclaim_narrow_sea_condition
			any_held_title = {
				title_capital_county.title_province = { geographical_region = gad_narrow_sea }
				tier = tier_kingdom
				count >= 3
				save_temporary_scope_as = de_jure_kingdom
			}
			root = { completely_controls = scope:de_jure_kingdom }
		}
		has_title = primary_title.title_capital_county
		is_imprisoned = no
	}

	ai_will_do = {
		base = 5
		modifier = {
			has_trait = ambitious
			add = 40
		}
		modifier = {
			has_trait = authoritative
			add = 5
		}
	}


	effect = {
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:gad_proclaim_narrow_sea_decision
		}
		save_scope_as = founder
		show_as_tooltip = { gad_proclaim_e_narrow_sea_effect = yes }
		trigger_event = { id = gad_title_creation_decisions.0003 }
	}

	cost = { gold = 3000 }
}
